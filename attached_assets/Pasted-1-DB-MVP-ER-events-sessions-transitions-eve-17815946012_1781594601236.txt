1. DB設計書 
まず、MVPではテーブルを増やしすぎない方が良いです。 
ER図 
events 
│ 
├── sessions 
│ 
└── transitions 
events 
生ログ保存 
最重要テーブル 
CREATE TABLE events ( 
id INTEGER PRIMARY KEY AUTOINCREMENT, 
timestamp DATETIME NOT NULL, 
event_type TEXT NOT NULL, 
app_name TEXT, 
window_title TEXT, 
domain TEXT, 
metadata TEXT 
); 
event_type 
window_changed 
mouse_active 
keyboard_active 
idle_start 
idle_end 
 
サンプル 
{ 
 "id": 1, 
 "timestamp": "2026-06-16T14:20:01", 
 "event_type": "window_changed", 
 "app_name": "chrome.exe", 
 "domain": "youtube.com" 
} 
 
sessions 
集計用 
CREATE TABLE sessions ( 
   id INTEGER PRIMARY KEY AUTOINCREMENT, 
 
   start_time DATETIME NOT NULL, 
   end_time DATETIME NOT NULL, 
 
   duration_seconds INTEGER, 
 
   app_name TEXT, 
   domain TEXT 
); 
 
サンプル 
{ 
 "app_name":"chrome.exe", 
 "domain":"youtube.com", 
 "duration_seconds":1800 
} 
 
transitions 
行動遷移分析 
CREATE TABLE transitions ( 
   id INTEGER PRIMARY KEY AUTOINCREMENT, 
 
   timestamp DATETIME, 
 
from_app TEXT, 
from_domain TEXT, 
to_app TEXT, 
to_domain TEXT 
); 
サンプル 
VSCode 
↓ 
GitHub 
{ 
"from_domain":"github.com", 
"to_domain":"youtube.com" 
} 
category_rules 
カテゴリ管理 
CREATE TABLE category_rules ( 
id INTEGER PRIMARY KEY AUTOINCREMENT, 
domain TEXT, 
category TEXT 
); 
例 
github.com 
→ 開発 
chatgpt.com 
→ 学習 
youtube.com 
→ 娯楽
